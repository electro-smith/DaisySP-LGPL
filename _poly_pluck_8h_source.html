58 float Process(float &trig, float note)
59 {
60 float sig, tval;
61 sig = 0.0f;
62 if(trig > 0.0f)
63 {
64 // increment active voice
65 active_voice_ = (active_voice_ + 1) % num_voices;
66 // set new voice to new note
67 plk_[active_voice_].SetDamp(p_damp_);
68 plk_[active_voice_].SetDecay(p_decay_);
69 plk_[active_voice_].SetAmp(0.25f);
70 }
71 plk_[active_voice_].SetFreq(mtof(note));
72
73 for(size_t i = 0; i < num_voices; i++)
74 {
75 tval = (trig > 0.0f && i == active_voice_) ? 1.0f : 0.0f;
76 sig += plk_[i].Process(tval);
77 }
78 if(trig > 0.0f)
79 trig = 0.0f;
80 return blk_.Process(sig);
81 }

86 void SetDecay(float p) { p_damp_ = p; }
